Câu lệnh switch case

    biểu thức thông thường là 1 biến trong js
    switch(biểu thức) {
        case x:
            // các câu lệnh case x
            break;
        case y:
            // các câu lệnh case y
            break;
        case z:
            // các câu lệnh case z
            break;
        case w:
            // các câu lệnh case w
            break;
        default:
            // các câu lệnh default
    }

    biểu thức trong () == x => code block x sẽ chạy
    biểu thức trong () == y => code block y sẽ chạy
    biểu thức trong () == z => code block z sẽ chạy
    biểu thức trong () == w => code block w sẽ chạy
    trong trường hợp biểu thức trong () không bằng bắt cứ 1 case nào
    thì code block trong default sẽ chạy

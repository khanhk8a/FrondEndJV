Câu lệnh điều else if

    if (điều kiện 1) {
        // điều kiện 1 trả về true => các câu lệnh trong đây sẽ chạy

    } else if (điều kiện 2) {
         // điều kiện 2 trả về true => các câu lệnh trong đây sẽ chạy

    } else if (điều kiện 3) {
        // điều kiện 3 trả về true => các câu lệnh trong đây sẽ chạy

    } else if (điều kiện 4) {
        // điều kiện 4 trả về true => các câu lệnh trong đây sẽ chạy

    } else {
        // khi mà điều kiện 1,2,3,4 đều trả về false thì các câu lệnh trong {} else sẽ chạy

    }

    Lưu ý : khi mà điều kiện 1 == true
    điều kiện 2 == true
    điều kiện 3 == true
    điều kiện 4 == true
    vậy thì điều kiện nào khi mà js chạy trước thì nó sẽ chọn các câu lệnh trong điều kiện đó chạy mà thôi
    ví dụ gặp đk 1 mà true thì chỉ thực hiện code trong {} của đk1
    nhưng khi mà đảo điều kiện 2 lên trước điều kiện 1 thì chỉ chạy điều 2 mà thôi
